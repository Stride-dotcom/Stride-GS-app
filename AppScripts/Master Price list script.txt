/**
 * Atomic counter increment via LockService.
 * Routes on payload.action:
 *   "getNextShipmentId" → SHP-000001 (Phase 1)
 *   "getNextInvoiceId"  → INV-000001 (Phase 2)
 *   (absent)            → defaults to getNextShipmentId for v1.x compat
 *
 * Returns JSON: { success: true,  shipmentNo|invoiceNo: "..." }
 *            or { success: false, error: "..." }
 */
function doPost(e) {
  if (!e || !e.postData || !e.postData.contents) {
    return createResponse_({ success: false, error: "Empty or malformed request body." });
  }

  let payload;
  try {
    payload = JSON.parse(e.postData.contents);
  } catch (_) {
    return createResponse_({ success: false, error: "Invalid JSON in request body." });
  }

  const ss            = getMasterSpreadsheetForRpc_();
  const expectedToken = getSetting_(ss, MPL_SETTINGS_KEYS.MASTER_RPC_TOKEN);

  if (!expectedToken) {
    return createResponse_({ success: false, error: "MASTER_RPC_TOKEN not configured on the Master sheet." });
  }

  if (String(payload.token || "").trim() !== expectedToken.trim()) {
    return createResponse_({ success: false, error: "Unauthorized." });
  }

  const lock = LockService.getScriptLock();
  try {
    lock.waitLock(30000);
  } catch (_) {
    return createResponse_({ success: false, error: "Lock timeout — too many concurrent requests." });
  }

  try {
    const settings = ss.getSheetByName(MPL_SH.SETTINGS);
    if (!settings) throw new Error("Settings sheet not found.");

    const data = settings.getDataRange().getValues();

    // Default to getNextShipmentId when action is absent (v1.x backward compat)
    const action = String(payload.action || "getNextShipmentId");

    if (action === "getNextShipmentId") {
      const rowIndex = data.findIndex(r => String(r[0]).trim() === MPL_SETTINGS_KEYS.GLOBAL_SHIPMENT_COUNTER);
      if (rowIndex === -1) throw new Error("GLOBAL_SHIPMENT_COUNTER key not found in Settings.");

      const currentVal = parseInt(data[rowIndex][1], 10) || 0;
      const nextVal    = currentVal + 1;
      settings.getRange(rowIndex + 1, 2).setValue(nextVal);
      SpreadsheetApp.flush();

      return createResponse_({
        success: true,
        id: "SHP-" + String(nextVal).padStart(6, "0"),
        shipmentNo: "SHP-" + String(nextVal).padStart(6, "0"),
      });
    }

    if (action === "getNextInvoiceId" || action === "getNextInvoiceNo" || action === "getNextInvoiceNumber") {
      const rowIndex = data.findIndex(r => String(r[0]).trim() === MPL_SETTINGS_KEYS.GLOBAL_INVOICE_COUNTER);
      if (rowIndex === -1) throw new Error("GLOBAL_INVOICE_COUNTER key not found in Settings.");

      const currentVal = parseInt(data[rowIndex][1], 10) || 0;
      const nextVal    = currentVal + 1;
      settings.getRange(rowIndex + 1, 2).setValue(nextVal);
      SpreadsheetApp.flush();

       return createResponse_({
        success: true,
        id: "INV-" + String(nextVal).padStart(6, "0"),        // Client looks for 'id'
        invoiceId: "INV-" + String(nextVal).padStart(6, "0"), // Client looks for 'invoiceId'
        invoiceNo: "INV-" + String(nextVal).padStart(6, "0"), // (Keep for safety)
      });
    }

    return createResponse_({ success: false, error: "Unknown action: " + action });

  } catch (err) {
    return createResponse_({ success: false, error: err.toString() });
  } finally {
    lock.releaseLock();
  }
}

/**
 * Web App-safe: opens the Master spreadsheet by ID stored in ScriptProperties.
 * - Primary:   ScriptProperties STRIDE_MASTER_SPREADSHEET_ID
 * - Secondary: bound container spreadsheet (getActiveSpreadsheet)
 * Throws if neither is available.
 */
function getMasterSpreadsheetForRpc_() {
  const props       = PropertiesService.getScriptProperties();
  const idFromProps = String(props.getProperty(MPL_PROPS.MASTER_SPREADSHEET_ID) || "").trim();
  if (idFromProps) return SpreadsheetApp.openById(idFromProps);

  const active = SpreadsheetApp.getActiveSpreadsheet();
  if (active) return active;

  throw new Error("Master spreadsheet ID not configured. Run StrideMasterPriceSetup() once.");
}

/**
 * Serialises a plain object to a JSON ContentService response.
 * @param {Object} obj
 * @return {GoogleAppsScript.Content.TextOutput}
 */
function createResponse_(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

/**
 * Reads a single setting value from the Settings sheet.
 * @param {GoogleAppsScript.Spreadsheet.Spreadsheet} ss
 * @param {string} key  The key name in column A of Settings
 * @return {string|undefined}
 */
function getSetting_(ss, key) {
  const sheet = ss.getSheetByName(MPL_SH.SETTINGS);
  if (!sheet) return undefined;
  const data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (String(data[i][0]).trim() === key) {
      return String(data[i][1]);
    }
  }
  return undefined;
}

/**
 * Web App GET endpoint for exporting Master data as JSON.
 *
 * URL format:
 *   ?token=...&action=exportAll
 *
 * Actions:
 *   exportAll (default)
 *   exportPriceList
 *   exportClassMap
 *   exportEmailTemplates
 *   exportInvoiceTemplates
 */
function doGet(e) {
  try {
    const params = (e && e.parameter) ? e.parameter : {};
    const token  = String(params.token || "").trim();
    const action = String(params.action || "").trim() || "exportAll";

    // Validate presence of token BEFORE opening any spreadsheet
    if (!token) {
      return createResponse_({ success: false, error: "Unauthorized." });
    }

    // Now open Master + validate shared secret
    const ss            = getMasterSpreadsheetForRpc_();
    const expectedToken = getSetting_(ss, MPL_SETTINGS_KEYS.MASTER_RPC_TOKEN);

    if (!expectedToken) {
      return createResponse_({ success: false, error: "MASTER_RPC_TOKEN not configured on the Master sheet." });
    }

    if (token !== expectedToken.trim()) {
      return createResponse_({ success: false, error: "Unauthorized." });
    }

    const out = exportMasterData_(ss, action);
    return createResponse_({ success: true, action: action, data: out });

  } catch (err) {
    const msg = (err && err.toString) ? err.toString() : String(err);
    return createResponse_({ success: false, error: msg });
  }
}

/**
 * Export router for Master data.
 * @param {GoogleAppsScript.Spreadsheet.Spreadsheet} ss
 * @param {string} action
 * @return {Object}
 */
function exportMasterData_(ss, action) {
  if (action === "exportAll") {
    return {
      version:         MPL_V,
      exportedAt:      new Date().toISOString(),
      priceList:       exportSheetAsRows_(ss, MPL_SH.PRICE),
      classMap:        exportSheetAsRows_(ss, MPL_SH.CLASS_MAP),
      emailTemplates:  exportTemplatesAsMap_(ss, MPL_SH.EMAIL_TEMPLATES),
      invoiceTemplates: exportTemplatesAsMap_(ss, MPL_SH.INVOICE_TEMPLATES),
    };
  }

  if (action === "exportPriceList") {
    return exportSheetAsRows_(ss, MPL_SH.PRICE);
  }
  if (action === "exportClassMap") {
    return exportSheetAsRows_(ss, MPL_SH.CLASS_MAP);
  }
  if (action === "exportEmailTemplates") {
    return exportTemplatesAsMap_(ss, MPL_SH.EMAIL_TEMPLATES);
  }
  if (action === "exportInvoiceTemplates") {
    return exportTemplatesAsMap_(ss, MPL_SH.INVOICE_TEMPLATES);
  }

  throw new Error("Unknown action: " + action);
}

/**
 * Exports a sheet as { headers: [...], rows: [...] } trimming trailing
 * empty columns and empty bottom rows.
 *
 * - headers: the first row after trimming trailing empty columns
 * - rows:    remaining rows trimmed to header width, excluding trailing
 *            fully-empty rows
 *
 * @param {GoogleAppsScript.Spreadsheet.Spreadsheet} ss
 * @param {string} sheetName
 * @return {{headers: string[], rows: any[][]}}
 */
function exportSheetAsRows_(ss, sheetName) {
  const sh = ss.getSheetByName(sheetName);
  if (!sh) throw new Error("Sheet not found: " + sheetName);

  const values = sh.getDataRange().getValues();
  if (!values || values.length === 0) return { headers: [], rows: [] };

  const headersRaw = values[0] || [];
  const headerLast = lastNonEmptyIndex_(headersRaw);
  const width      = Math.max(0, headerLast + 1);

  const headers = headersRaw.slice(0, width).map(h => String(h || "").trim());
  const rowsRaw = values.slice(1);

  // Trim bottom fully-empty rows
  let lastRowIdx = -1;
  for (let i = rowsRaw.length - 1; i >= 0; i--) {
    const row = rowsRaw[i] || [];
    const hasAny = row.slice(0, width).some(c => String(c) !== "");
    if (hasAny) { lastRowIdx = i; break; }
  }
  const trimmedRows = lastRowIdx === -1 ? [] : rowsRaw.slice(0, lastRowIdx + 1);

  // Trim each row to the header width
  const rows = trimmedRows.map(r => (r || []).slice(0, width));

  return { headers: headers, rows: rows };
}

/**
 * Exports a template sheet (Email_Templates or Invoice_Templates) as:
 *   { "<TEMPLATE_KEY>": { subject: "...", body: "...", notes: "..." }, ... }
 *
 * Assumes schema:
 *   A: Template Key
 *   B: Subject
 *   C: HTML Body
 *   D: Notes / Token Reference
 *
 * @param {GoogleAppsScript.Spreadsheet.Spreadsheet} ss
 * @param {string} sheetName
 * @return {Object.<string,{subject:string,body:string,notes:string}>}
 */
function exportTemplatesAsMap_(ss, sheetName) {
  const sh = ss.getSheetByName(sheetName);
  if (!sh) throw new Error("Sheet not found: " + sheetName);

  const values = sh.getDataRange().getValues();
  if (!values || values.length < 2) return {};

  const out = {};
  for (let i = 1; i < values.length; i++) {
    const row = values[i] || [];
    const key = String(row[0] || "").trim();
    if (!key) continue;

    out[key] = {
      subject: String(row[1] || ""),
      body:    String(row[2] || ""),
      notes:   String(row[3] || ""),
    };
  }
  return out;
}

/**
 * Returns the last index in an array where the cell is non-empty
 * (by String(cell) !== ""). Returns -1 if all empty.
 * @param {Array} arr
 * @return {number}
 */
function lastNonEmptyIndex_(arr) {
  if (!arr || !arr.length) return -1;
  for (let i = arr.length - 1; i >= 0; i--) {
    if (String(arr[i]) !== "") return i;
  }
  return -1;
}

/**
 * Reads the Settings sheet into a key->value map.
 * Column A = key, Column B = value. Skips header row.
 *
 * @param  {GoogleAppsScript.Spreadsheet.Sheet} settingsSheet
 * @return {Object.<string, string>}
 */
function readSettingsMap_(settingsSheet) {
  const map = {};
  if (!settingsSheet) return map;

  const data = settingsSheet.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    const key = String(data[i][0] || "").trim();
    if (key) {
      map[key] = String(data[i][1] || "");
    }
  }
  return map;
}


/**
 * Tries to get the current user email via Session.getActiveUser().
 * Returns empty string if unavailable (common in Web App / trigger context).
 *
 * @return {string}
 */
function tryGetEmail_() {
  try {
    const email = Session.getActiveUser().getEmail();
    return email || "";
  } catch (e) {
    return "";
  }
}


/**
 * Parses a value as a non-negative integer.
 * Returns fallback if the value is empty, NaN, negative, or not finite.
 *
 * @param  {*}      val
 * @param  {number} fallback
 * @return {number}
 */
function toNonNegativeInt_(val, fallback) {
  if (val === undefined || val === null || String(val).trim() === "") {
    return fallback;
  }
  const n = parseInt(val, 10);
  if (isNaN(n) || !isFinite(n) || n < 0) {
    return fallback;
  }
  return n;
}


/**
 * Applies owner-only SHEET-level protections to every sheet.
 * Uses OWNER_EMAIL from Settings as the sole editor.
 *
 * Orphan guard: if OWNER_EMAIL is blank or the active user cannot be
 * determined, skips protection to avoid locking yourself out.
 *
 * PUBLIC function (no trailing underscore) - exposed as menu item.
 */
function StrideMPL_ApplyOwnerProtections() {
  const ss = SpreadsheetApp.getActive();
  const ownerEmail = getSetting_(ss, MPL_SETTINGS_KEYS.OWNER_EMAIL);

  if (!ownerEmail) {
    SpreadsheetApp.getUi().alert(
      "OWNER_EMAIL is not set in the Settings sheet.\n" +
      "Skipping protections to avoid lockout.\n" +
      "Set OWNER_EMAIL first, then re-run."
    );
    return;
  }

  let activeEmail = "";
  try { activeEmail = Session.getActiveUser().getEmail(); } catch (e) { /* expected in some contexts */ }
  if (!activeEmail) {
    SpreadsheetApp.getUi().alert(
      "Cannot determine your email. Skipping protections to avoid lockout.\n" +
      "Run this from the spreadsheet UI (not a trigger)."
    );
    return;
  }

  const sheets = ss.getSheets();
  for (let s = 0; s < sheets.length; s++) {
    const sheet = sheets[s];
    const protections = sheet.getProtections(SpreadsheetApp.ProtectionType.SHEET);
    for (let p = 0; p < protections.length; p++) {
      if (protections[p].canEdit()) {
        protections[p].remove();
      }
    }

    const protection = sheet.protect()
      .setDescription("Owner only - " + sheet.getName());
    protection.addEditor(ownerEmail);
    if (activeEmail.toLowerCase() !== ownerEmail.toLowerCase()) {
      protection.addEditor(activeEmail);
    }
    // Collect editors to remove first, then remove (avoid modifying list during iteration)
    const editors = protection.getEditors();
    const toRemove = [];
    for (let e = 0; e < editors.length; e++) {
      const editorEmail = editors[e].getEmail();
      if (editorEmail.toLowerCase() !== ownerEmail.toLowerCase() &&
          editorEmail.toLowerCase() !== activeEmail.toLowerCase()) {
        toRemove.push(editors[e]);
      }
    }
    for (let r = 0; r < toRemove.length; r++) {
      protection.removeEditor(toRemove[r]);
    }
    protection.setDomainEdit(false);
  }

  SpreadsheetApp.getUi().alert(
    "Owner protections applied to all " + sheets.length + " sheets.\n" +
    "Owner: " + ownerEmail
  );
}


/**
 * Ensures a sheet with the given name exists in the spreadsheet.
 * If it does not exist, creates it. Returns the sheet either way.
 *
 * @param  {GoogleAppsScript.Spreadsheet.Spreadsheet} ss
 * @param  {string} name
 * @return {GoogleAppsScript.Spreadsheet.Sheet}
 */
function ensureSheet_(ss, name) {
  const sheet = ss.getSheetByName(name);
  if (sheet) return sheet;
  return ss.insertSheet(name);
}


/**
 * Ensures row 1 of the given sheet matches the expected header array.
 * - If row 1 is completely empty, writes the headers.
 * - If row 1 has data but does not match, overwrites ONLY row 1
 *   (repairs headers without touching data rows).
 *
 * @param {GoogleAppsScript.Spreadsheet.Sheet} sheet
 * @param {string[]} headers  Expected header values
 */
function ensureHeaderRow_(sheet, headers) {
  if (!headers || headers.length === 0) return;

  let currentRow = [];
  if (sheet.getLastRow() >= 1) {
    const colsToRead = Math.max(headers.length, Math.min(sheet.getMaxColumns(), headers.length + 2));
    currentRow = sheet.getRange(1, 1, 1, colsToRead).getValues()[0];
  }

  let needsRepair = false;
  if (currentRow.length < headers.length) {
    needsRepair = true;
  } else {
    for (let i = 0; i < headers.length; i++) {
      if (String(currentRow[i] || "").trim() !== String(headers[i]).trim()) {
        needsRepair = true;
        break;
      }
    }
  }

  if (needsRepair) {
    if (sheet.getMaxColumns() < headers.length) {
      sheet.insertColumnsAfter(sheet.getMaxColumns(),
        headers.length - sheet.getMaxColumns());
    }
    sheet.getRange(1, 1, 1, headers.length)
      .setValues([headers])
      .setFontWeight("bold");
  }
}


/**
 * Returns true if the sheet has any non-empty data below row 1.
 * Used to decide whether to seed default data - we NEVER overwrite
 * existing data (per the design philosophy).
 *
 * @param  {GoogleAppsScript.Spreadsheet.Sheet} sheet
 * @return {boolean}
 */
function hasNonHeaderData_(sheet) {
  const lastRow = sheet.getLastRow();
  if (lastRow <= 1) return false;

  const lastCol = Math.max(sheet.getLastColumn(), 1);
  const values = sheet.getRange(2, 1, lastRow - 1, lastCol).getValues();

  for (let r = 0; r < values.length; r++) {
    for (let c = 0; c < values[r].length; c++) {
      if (String(values[r][c]).trim() !== "") {
        return true;
      }
    }
  }
  return false;
}

/* ============================================================
   SYNC TO ALL CLIENTS — v2.2.0
   Pushes Price_List, Class_Map, and Email_Templates data to
   every active client's local cache tabs (Price_Cache, Class_Cache,
   Email_Template_Cache). Reads client list from CB Clients tab.
   ============================================================ */
function StrideMPL_SyncToAllClients() {
  var ui = SpreadsheetApp.getUi();
  var ss = SpreadsheetApp.getActive();

  var cbId = getSetting_(ss, MPL_SETTINGS_KEYS.CB_SPREADSHEET_ID);
  if (!cbId) {
    ui.alert("Missing CB_SPREADSHEET_ID in Settings.\n\nAdd a row with Key = 'CB_SPREADSHEET_ID' and Value = your Consolidated Billing spreadsheet ID.");
    return;
  }

  // Read source data from this Master spreadsheet
  var priceSh = ss.getSheetByName(MPL_SH.PRICE);
  var classSh = ss.getSheetByName(MPL_SH.CLASS_MAP);
  var emailSh = ss.getSheetByName(MPL_SH.EMAIL_TEMPLATES);

  if (!priceSh || !classSh) { ui.alert("Missing Price_List or Class_Map sheet."); return; }

  var priceData = priceSh.getDataRange().getValues();
  var classData = classSh.getDataRange().getValues();
  var emailData = emailSh ? emailSh.getDataRange().getValues() : null;

  // Get active clients from CB Clients tab
  var cbSs;
  try { cbSs = SpreadsheetApp.openById(cbId); } catch (e) {
    ui.alert("Could not open Consolidated Billing sheet.\n\n" + e);
    return;
  }

  var clientsSh = cbSs.getSheetByName("Clients");
  if (!clientsSh || clientsSh.getLastRow() < 2) { ui.alert("No Clients tab or no client rows in CB sheet."); return; }

  var cData = clientsSh.getDataRange().getValues();
  var cHeaders = {};
  cData[0].forEach(function(h, i) { cHeaders[String(h || "").trim().toUpperCase()] = i; });

  var nameIdx = cHeaders["CLIENT NAME"];
  var idIdx = cHeaders["CLIENT SPREADSHEET ID"];
  var activeIdx = cHeaders["ACTIVE"];

  if (idIdx === undefined) { ui.alert("Clients tab missing 'Client Spreadsheet ID' column."); return; }

  var clients = [];
  for (var i = 1; i < cData.length; i++) {
    var cId = String(cData[i][idIdx] || "").trim();
    var cName = nameIdx !== undefined ? String(cData[i][nameIdx] || "").trim() : "Client " + i;
    var cActive = activeIdx !== undefined ? String(cData[i][activeIdx] || "").trim().toUpperCase() : "TRUE";
    if (cId && (cActive === "TRUE" || cActive === "YES" || cActive === "Y" || cActive === "1")) {
      clients.push({ name: cName, id: cId });
    }
  }

  if (!clients.length) { ui.alert("No active clients found."); return; }

  var confirm = ui.alert(
    "Sync to All Clients",
    "This will push Price_Cache, Class_Cache, and Email_Template_Cache to " + clients.length + " active client(s).\n\nContinue?",
    ui.ButtonSet.YES_NO
  );
  if (confirm !== ui.Button.YES) return;

  var success = 0, failed = [];
  for (var c = 0; c < clients.length; c++) {
    try {
      var clientSs = SpreadsheetApp.openById(clients[c].id);

      // Price_Cache
      var pcSh = clientSs.getSheetByName("Price_Cache") || clientSs.insertSheet("Price_Cache");
      MPL_writeCache_(pcSh, priceData, "Price_Cache");

      // Class_Cache
      var ccSh = clientSs.getSheetByName("Class_Cache") || clientSs.insertSheet("Class_Cache");
      MPL_writeCache_(ccSh, classData, "Class_Cache");

      // Email_Template_Cache
      if (emailData) {
        var etSh = clientSs.getSheetByName("Email_Template_Cache") || clientSs.insertSheet("Email_Template_Cache");
        MPL_writeCache_(etSh, emailData, "Email_Template_Cache");
      }

      success++;
      SpreadsheetApp.getActive().toast("Synced " + (c + 1) + "/" + clients.length + ": " + clients[c].name, "Sync Progress", 2);
    } catch (err) {
      failed.push(clients[c].name + " (" + err.message + ")");
      Logger.log("Sync failed for " + clients[c].name + ": " + err);
    }
  }

  var msg = "Sync complete.\n\nSuccess: " + success + " / " + clients.length;
  if (failed.length) {
    msg += "\n\nFailed:\n• " + failed.slice(0, 15).join("\n• ");
    if (failed.length > 15) msg += "\n• ... (" + (failed.length - 15) + " more)";
  }
  ui.alert(msg);
}

/** Writes source data to a cache sheet (clear + write + format header). */
function MPL_writeCache_(dstSheet, srcData, cacheName) {
  dstSheet.clearContents();
  dstSheet.clearFormats();
  if (!srcData || srcData.length < 1) {
    dstSheet.getRange(1, 1).setValue(cacheName + " (empty)");
    return;
  }
  dstSheet.getRange(1, 1, srcData.length, srcData[0].length).setValues(srcData);
  try {
    dstSheet.getRange(1, 1, 1, srcData[0].length)
      .setFontWeight("bold")
      .setBackground("#0F172A")
      .setFontColor("#ffffff");
  } catch (_) {}
  dstSheet.setFrozenRows(1);
}
